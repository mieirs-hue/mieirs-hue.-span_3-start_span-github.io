    <script>
        let scrollInterval = null;

        // 🔽 Engine 1: Smoothly steps the screen downward 40%
        function triggerSlowEscalatorDown() {
            clearInterval(scrollInterval);
            scrollInterval = setInterval(() => {
                const currentScroll = window.scrollY;
                const maxScroll = document.body.scrollHeight * 0.4; // Steps down 40% of the page
                
                if (currentScroll < maxScroll) {
                    window.scrollBy(0, 6); // Controls the speed (6 is a smooth glide)
                } else {
                    clearInterval(scrollInterval);
                }
            }, 10);
        }

        // ▲ Engine 2: Smoothly steps the screen back up to the top root
        function triggerSlowEscalatorUp() {
            clearInterval(scrollInterval);
            scrollInterval = setInterval(() => {
                const currentScroll = window.scrollY;
                
                if (currentScroll > 0) {
                    window.scrollBy(0, -6); // Smooth glide upward (-6 steps backward)
                } else {
                    clearInterval(scrollInterval);
                }
            }, 10);
        }

        // 🔄 Automatic Swap Loop: Changes the HOVER behavior flawlessly based on scroll position
        window.addEventListener('scroll', () => {
            const textNode = document.getElementById('escalatorText');
            const arrowNode = document.getElementById('escalatorArrow');
            const nodeBox = document.getElementById('escalatorNode');

            if (!textNode || !arrowNode || !nodeBox) return;

            if (window.scrollY > 150) {
                // 🚀 STATE 1: User scrolled down -> Change hover to slide BACK UP
                textNode.innerText = "[ HOVER TO ESCALATE BACK TO TOP ]";
                arrowNode.innerText = "▲";
                
                nodeBox.removeAttribute('onclick'); // Guard rail
                nodeBox.onmouseenter = () => triggerSlowEscalatorUp(); // Sets hover engine up
            } else {
                // 🛬 STATE 2: User at the top -> Change hover to slide DOWN
                textNode.innerText = "[ HOVER TO ESCALATE DOWN ]";
                arrowNode.innerText = "▼";
                
                nodeBox.removeAttribute('onclick'); // Guard rail
                nodeBox.onmouseenter = () => triggerSlowEscalatorDown(); // Sets hover engine down
            }
        });
    </script>
